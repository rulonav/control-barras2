// src/screens/MainMenuScreen.js
import React, { useEffect, useState } from 'react';
import { 
  View, 
  Text, 
  Alert, 
  BackHandler,
  TextInput,
  Modal,
  TouchableOpacity,
  KeyboardAvoidingView,
  Platform,
  ScrollView
} from 'react-native';
import { Card, Title, Button, Paragraph, Divider } from 'react-native-paper';
import databaseService from '../services/databaseService';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { styles } from '../styles/MainMenuScreenStyles';

const MainMenuScreen = ({ navigation }) => {
  const [usuario, setUsuario] = useState(null);
  const [loading, setLoading] = useState(true);
  
  // Estados para el modal de configuración de rango
  const [modalRangoVisible, setModalRangoVisible] = useState(false);
  const [codigoInicial, setCodigoInicial] = useState('');
  const [codigoFinal, setCodigoFinal] = useState('');
  const [tipoRuta, setTipoRuta] = useState('numerica');
  const [nombreRuta, setNombreRuta] = useState('');

  useEffect(() => {
    cargarUsuarioYVerificarRuta();
    
    const backAction = () => {
      Alert.alert(
        'Salir de la App',
        '¿Estás seguro que deseas salir?',
        [
          { text: 'Cancelar', style: 'cancel' },
          { text: 'Salir', style: 'destructive', onPress: () => BackHandler.exitApp() }
        ]
      );
      return true;
    };
    
    const backHandler = BackHandler.addEventListener('hardwareBackPress', backAction);
    
    return () => backHandler.remove();
  }, []);

  const cargarUsuarioYVerificarRuta = async () => {
    try {
      const userData = await AsyncStorage.getItem('userData');
      if (!userData) {
        console.log('⚠️ No hay usuario en AsyncStorage, volviendo a Login');
        navigation.replace('Login');
        return;
      }
      
      const usuarioParseado = JSON.parse(userData);
      setUsuario(usuarioParseado);
      
      if (!usuarioParseado.id) {
        console.log('⚠️ Usuario sin ID válido, volviendo a Login');
        navigation.replace('Login');
        return;
      }
      
      console.log('✅ Usuario cargado con ID:', usuarioParseado.id);
      
      try {
        const rutasActivas = await databaseService.obtenerRutasUsuario(usuarioParseado.id);
        const rutaActiva = rutasActivas?.find(r => !r.finalizada);
        
        if (rutaActiva) {
          console.log('🔄 Ruta activa encontrada:', rutaActiva.numero);
          Alert.alert(
            'Ruta Incompleta',
            `Hay una ruta activa: ${rutaActiva.numero}. ¿Deseas continuar?`,
            [
              {
                text: 'Nueva Ruta',
                style: 'cancel',
                onPress: () => setLoading(false)
              },
              {
                text: 'Continuar',
                onPress: () => {
                  navigation.navigate('Scanner', {
                    rutaRecuperada: rutaActiva,
                    tieneBuffer: true
                  });
                }
              }
            ]
          );
        }
      } catch (error) {
        console.error('❌ Error al verificar ruta activa:', error);
      }
    } catch (error) {
      console.error('❌ Error cargando usuario:', error);
      navigation.replace('Login');
    } finally {
      setLoading(false);
    }
  };

  // ✅ VALIDAR RANGO DE CÓDIGOS
  const validarRango = () => {
    const inicial = parseInt(codigoInicial.replace(/\D/g, ''), 10);
    const final = parseInt(codigoFinal.replace(/\D/g, ''), 10);
    
    if (!codigoInicial.trim() || !codigoFinal.trim()) {
      Alert.alert('Error', 'Debes ingresar ambos códigos del rango');
      return false;
    }
    
    if (isNaN(inicial) || isNaN(final)) {
      Alert.alert('Error', 'Los códigos deben ser numéricos');
      return false;
    }
    
    if (inicial > 44000000000 && final > 44000000000) {
      if (inicial >= final) {
        Alert.alert('Error', 'El código inicial debe ser menor al código final');
        return false;
      }
      return true;
    }
    
    // Para rutas numéricas o destinadas
    if (inicial <= 0 || final <= 0) {
      Alert.alert('Error', 'Los números de ruta deben ser positivos');
      return false;
    }
    
    if (inicial >= final) {
      Alert.alert('Error', 'El número inicial debe ser menor al número final');
      return false;
    }
    
    return true;
  };

  // ✅ MOSTRAR POPUP DE CONFIRMACIÓN ANTES DE ESCANEAR
  const confirmarInicioEscaneo = () => {
    if (!validarRango()) return;
    
    const inicialLimpio = codigoInicial.replace(/\D/g, '');
    const finalLimpio = codigoFinal.replace(/\D/g, '');
    
    Alert.alert(
      '⚠️ Confirmar Inicio de Escaneo',
      `¿Estás seguro que deseas iniciar el escaneo con los siguientes parámetros?
      
📋 Configuración:
• Tipo de Ruta: ${tipoRuta === 'numerica' ? 'Número de Ruta' : tipoRuta === 'destinada' ? 'Ruta Destinada' : 'Camión Precinto'}
• ${tipoRuta === 'numerica' ? 'Ruta' : 'Nombre'}: ${nombreRuta || 'Sin especificar'}
• Código Inicial: ${inicialLimpio}
• Código Final: ${finalLimpio}

✅ Una vez iniciado, deberás completar el escaneo o finalizar manualmente.`,
      [
        {
          text: '❌ Cancelar',
          style: 'cancel',
          onPress: () => console.log('Escaneo cancelado por usuario')
        },
        {
          text: '✅ Confirmar y Escanear',
          style: 'default',
          onPress: () => iniciarEscaneoConRango()
        }
      ],
      { cancelable: false }
    );
  };

  // ✅ INICIAR ESCANEO CON RANGO CONFIGURADO
  const iniciarEscaneoConRango = async () => {
    try {
      if (!usuario?.id) {
        Alert.alert('Error', 'Usuario no válido');
        return;
      }
      
      const inicialLimpio = codigoInicial.replace(/\D/g, '');
      const finalLimpio = codigoFinal.replace(/\D/g, '');
      
      // Preparar datos de la ruta con el rango
      const rutaData = {
        numero: nombreRuta.trim() || `${inicialLimpio}-${finalLimpio}`,
        usuario_id: usuario.id,
        rango: {
          inicial: inicialLimpio,
          final: finalLimpio,
          tipo: tipoRuta
        },
        tipo: tipoRuta
      };
      
      console.log('🚀 Iniciando escaneo con rango:', rutaData);
      
      // Cerrar modal
      setModalRangoVisible(false);
      
      // Navegar a ScannerScreen con los parámetros del rango
      navigation.navigate('Scanner', {
        rutaConfig: rutaData,
        rangoEscaneo: {
          inicial: inicialLimpio,
          final: finalLimpio,
          tipo: tipoRuta
        }
      });
      
      // Limpiar campos después de iniciar
      limpiarCamposRango();
      
    } catch (error) {
      console.error('❌ Error al iniciar escaneo:', error);
      Alert.alert('Error', 'No se pudo iniciar el escaneo: ' + error.message);
    }
  };

  // ✅ LIMPIAR CAMPOS DEL MODAL
  const limpiarCamposRango = () => {
    setCodigoInicial('');
    setCodigoFinal('');
    setNombreRuta('');
    setTipoRuta('numerica');
  };

  // ✅ ABRIR MODAL DE CONFIGURACIÓN DE RANGO
  const abrirModalConfigurarRango = () => {
    limpiarCamposRango();
    setModalRangoVisible(true);
  };

  const handleNuevaRuta = () => {
    // En lugar de navegar directamente, mostrar modal de configuración
    abrirModalConfigurarRango();
  };

  const handleHistorial = () => navigation.navigate('Historial');
  
  const handleControlListado = () => navigation.navigate('ControlList');
  
  const handleLogout = async () => {
    Alert.alert(
      'Cerrar Sesión',
      '¿Estás seguro que deseas cerrar sesión?',
      [
        { text: 'Cancelar', style: 'cancel' },
        {
          text: 'Salir',
          style: 'destructive',
          onPress: async () => {
            await AsyncStorage.removeItem('userData');
            navigation.replace('Login');
          }
        }
      ]
    );
  };

  // ✅ RENDERIZAR MODAL DE CONFIGURACIÓN DE RANGO
  const renderModalConfigurarRango = () => (
    <Modal
      visible={modalRangoVisible}
      animationType="slide"
      transparent={true}
      onRequestClose={() => setModalRangoVisible(false)}
    >
      <KeyboardAvoidingView
        behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
        style={styles.modalOverlay}
      >
        <View style={styles.modalContent}>
          <Title style={styles.modalTitle}>⚙️ Configurar Rango de Escaneo</Title>
          <Paragraph style={styles.modalSubtitle}>
            Define los parámetros para tu nueva ruta de escaneo
          </Paragraph>
          
          <ScrollView style={styles.modalScroll}>
            {/* Selector de Tipo de Ruta */}
            <Text style={styles.inputLabel}>Tipo de Ruta *</Text>
            <View style={styles.tipoSelector}>
              <TouchableOpacity
                style={[
                  styles.tipoButton,
                  tipoRuta === 'destinada' && styles.tipoButtonActive
                ]}
                onPress={() => setTipoRuta('destinada')}
              >
                <Text style={styles.tipoButtonText}>Ruta Destinada</Text>
              </TouchableOpacity>
              <TouchableOpacity
                style={[
                  styles.tipoButton,
                  tipoRuta === 'numerica' && styles.tipoButtonActive
                ]}
                onPress={() => setTipoRuta('numerica')}
              >
                <Text style={styles.tipoButtonText}>Número de Ruta</Text>
              </TouchableOpacity>
              <TouchableOpacity
                style={[
                  styles.tipoButton,
                  tipoRuta === 'precinto' && styles.tipoButtonActive
                ]}
                onPress={() => setTipoRuta('precinto')}
              >
                <Text style={styles.tipoButtonText}>Camión Precinto</Text>
              </TouchableOpacity>
            </View>
            
            {/* Campo Nombre/Número de Ruta */}
            <Text style={styles.inputLabel}>
              {tipoRuta === 'numerica' 
                ? 'Número de Ruta (se agregará AM1__)' 
                : tipoRuta === 'destinada' 
                  ? 'Código de Ruta (Ej: KD1, A321)' 
                  : 'Número de Precinto'}
            </Text>
            <TextInput
              style={styles.input}
              value={nombreRuta}
              onChangeText={setNombreRuta}
              placeholder={
                tipoRuta === 'destinada'
                  ? 'Ej: KD1, A321, L9'
                  : tipoRuta === 'numerica'
                    ? 'Ej: 12 (se guardará como AM1__12)'
                    : 'Hasta 9 dígitos'
              }
              autoCapitalize={tipoRuta === 'destinada' ? 'characters' : 'none'}
              keyboardType={tipoRuta === 'destinada' ? 'default' : 'numeric'}
            />
            
            <Divider style={styles.divider} />
            
            {/* Campos de Rango de Códigos */}
            <Text style={styles.inputLabel}>📦 Rango de Códigos a Escanear *</Text>
            
            <Text style={styles.subLabel}>Código Inicial</Text>
            <TextInput
              style={styles.input}
              value={codigoInicial}
              onChangeText={setCodigoInicial}
              placeholder="Ej: 45000000001"
              keyboardType="numeric"
              maxLength={11}
            />
            
            <Text style={styles.subLabel}>Código Final</Text>
            <TextInput
              style={styles.input}
              value={codigoFinal}
              onChangeText={setCodigoFinal}
              placeholder="Ej: 45000000100"
              keyboardType="numeric"
              maxLength={11}
            />
            
            {/* Preview del Rango */}
            {(codigoInicial || codigoFinal) && (
              <View style={styles.previewContainer}>
                <Text style={styles.previewText}>
                  🔍 Rango configurado:
                </Text>
                <Text style={styles.previewValue}>
                  {codigoInicial.replace(/\D/g, '') || '___'} → {codigoFinal.replace(/\D/g, '') || '___'}
                </Text>
                <Text style={styles.previewCount}>
                  ~{Math.max(0, parseInt(codigoFinal.replace(/\D/g, '') || '0') - parseInt(codigoInicial.replace(/\D/g, '') || '0'))} paquetes estimados
                </Text>
              </View>
            )}
            
            <Text style={styles.helpText}>
              💡 Los códigos deben ser mayores a 44000000000 y tener 11 dígitos
            </Text>
          </ScrollView>
          
          {/* Botones del Modal */}
          <View style={styles.modalButtons}>
            <Button
              mode="outlined"
              onPress={() => {
                setModalRangoVisible(false);
                limpiarCamposRango();
              }}
              style={styles.modalButton}
              textColor="#f44336"
            >
              ❌ Cancelar
            </Button>
            <Button
              mode="contained"
              onPress={confirmarInicioEscaneo}
              style={[styles.modalButton, styles.confirmButton]}
              disabled={!codigoInicial.trim() || !codigoFinal.trim()}
            >
              ✅ Confirmar y Escanear
            </Button>
          </View>
        </View>
      </KeyboardAvoidingView>
    </Modal>
  );

  if (loading) {
    return (
      <View style={styles.loadingContainer}>
        <Text style={styles.loadingText}>Cargando menú...</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      {/* Header con información del usuario */}
      {usuario && (
        <Card style={styles.userCard}>
          <Card.Content>
            <View style={styles.userInfo}>
              <View style={styles.avatar}>
                <Text style={styles.avatarText}>
                  {usuario.nombre?.charAt(0)?.toUpperCase() || 'U'}
                </Text>
              </View>
              <View style={styles.userDetails}>
                <Text style={styles.userName}>
                  {usuario.nombre} {usuario.apellido}
                </Text>
                <Text style={styles.userStation}>
                  🏢 {usuario.estacion}
                </Text>
              </View>
            </View>
          </Card.Content>
        </Card>
      )}
      
      <Title style={styles.screenTitle}>Opciones Principales</Title>
      
      {/* Card: Escanear Nueva Ruta con Configuración */}
      <Card style={styles.menuCard}>
        <Card.Content style={styles.cardContentRow}>
          <View style={styles.cardIconContainer}>
            <Text style={styles.cardIcon}>📦</Text>
          </View>
          <View style={styles.cardTextContainer}>
            <Title style={styles.cardTitle}>Escanear Nueva Ruta</Title>
            <Paragraph style={styles.cardDescription}>
              Configurar rango de códigos y comenzar nuevo control de ruta
            </Paragraph>
          </View>
        </Card.Content>
        <Card.Actions style={styles.cardActions}>
          <Button
            mode="contained"
            onPress={handleNuevaRuta}
            style={styles.actionButton}
            icon="cog"
          >
            ⚙️ Configurar y Escanear
          </Button>
        </Card.Actions>
      </Card>
      
      {/* Card: Historial */}
      <Card style={styles.menuCard}>
        <Card.Content style={styles.cardContentRow}>
          <View style={styles.cardIconContainer}>
            <Text style={styles.cardIcon}>📋</Text>
          </View>
          <View style={styles.cardTextContainer}>
            <Title style={styles.cardTitle}>Historial de Rutas</Title>
            <Paragraph style={styles.cardDescription}>
              Ver rutas completadas, compartir reportes o eliminar registros
            </Paragraph>
          </View>
        </Card.Content>
        <Card.Actions style={styles.cardActions}>
          <Button
            mode="outlined"
            onPress={handleHistorial}
            style={styles.actionButton}
            icon="history"
          >
            Ver Historial
          </Button>
        </Card.Actions>
      </Card>
      
      {/* Card: Controlar Listado */}
      <Card style={styles.menuCard}>
        <Card.Content style={styles.cardContentRow}>
          <View style={styles.cardIconContainer}>
            <Text style={styles.cardIcon}>📊</Text>
          </View>
          <View style={styles.cardTextContainer}>
            <Title style={styles.cardTitle}>Controlar Listado</Title>
            <Paragraph style={styles.cardDescription}>
              Comparar escaneos con archivos CSV/XLSX de ruteo primario
            </Paragraph>
          </View>
        </Card.Content>
        <Card.Actions style={styles.cardActions}>
          <Button
            mode="outlined"
            onPress={handleControlListado}
            style={styles.actionButton}
            icon="file-compare"
          >
            Controlar
          </Button>
        </Card.Actions>
      </Card>
      
      {/* Botón de Cerrar Sesión */}
      <Button
        mode="outlined"
        onPress={handleLogout}
        style={styles.logoutButton}
        textColor="#f44336"
        icon="logout"
      >
        🚪 Cerrar Sesión
      </Button>
      
      {/* Modal de Configuración de Rango */}
      {renderModalConfigurarRango()}
    </View>
  );
};

export default MainMenuScreen;
